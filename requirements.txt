import socket
import psutil
import platform
import wmi  # pip install wmi

def get_system_info():
    # Initialize WMI
    c = wmi.WMI()
    
    # Get FQDN (Fully Qualified Domain Name)
    fqdn = socket.getfqdn()
    
    # Get IP Address (primary)
    ip_address = socket.gethostbyname(socket.gethostname())
    
    # Get OS information using platform
    os_name = platform.system()
    os_build = platform.version()
    
    # Alternative OS info using WMI (more detailed for Windows)
    try:
        for os in c.Win32_OperatingSystem():
            os_name = os.Caption
            os_build = os.BuildNumber
            break
    except:
        pass
    
    # Get CPU cores (physical and logical)
    cpu_cores = psutil.cpu_count(logical=False)  # physical cores
    logical_cores = psutil.cpu_count(logical=True)  # logical cores
    
    # Get total RAM
    total_ram = psutil.virtual_memory().total
    total_ram_gb = round(total_ram / (1024 ** 3), 2)  # Convert to GB
    
    # Print all information
    print(f"FQDN: {fqdn}")
    print(f"IP Address: {ip_address}")
    print(f"OS Name: {os_name}")
    print(f"OS Build: {os_build}")
    print(f"Physical CPU Cores: {cpu_cores}")
    print(f"Logical CPU Cores: {logical_cores}")
    print(f"Total RAM: {total_ram_gb} GB")

if __name__ == "__main__":
    get_system_info()
